1.暴力破解
#include<iostream>
using namespace std;
int MaxSubArray(int* a, int n)
{
	int maxSum = a[0];
	int currSum;
	for (int i = 0; i < n;i++)
	{
		for (int j = i; j < n;j++)
		{
			currSum = 0;
			for (int k = i; k <= j;k++)
			{
				currSum += a[k];
			}
			if (currSum>maxSum)
			{
				maxSum = currSum;
			}
		}
	}
	return maxSum;
}
int main()
{
	int ary1[10] = {-5,3,3,2,-3,4,5,7,-9,8};
	int a = MaxSubArray(ary1, 10);
	cout << a << endl;
	cin.get(); 
	return 0;
}
