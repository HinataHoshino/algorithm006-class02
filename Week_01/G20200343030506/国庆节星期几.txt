#include <iostream>
using namespace std;
int main()
{
	int  m = 0, count = 0;
	int week[7] = { 6, 7, 1, 2, 3, 4, 5 };//星期多少
	int month[12] = { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
	int monthday[12][31] = { 0 };//每天是周几
	int  k = 0;
	for (int yearname = 1949; yearname <= 2012; yearname++)
	{
		if (yearname % 4 == 0 && yearname % 100 != 0 || yearname % 400 == 0)
		{
			month[1] = 29;
		}
		else
		{
			month[1] = 28;
		}
		int b = 0;
		for (int i = 0; i < 12; i++)
		{
			if (yearname==1949&&b==0)
			{
				i = 9;
				b++;
			}
			for (int j = 0; j < month[i]; j++)
			{
				monthday[i][j] = week[k];
				k++;
				if (k == 7)
				{
					k = 0;
				}
				if (i == 9 && j == 0 && monthday[i][j] == 7)
				{
					count++;
				}
			}
		}
	}
	cout << count;
	cin.get();
	return 0;
}
