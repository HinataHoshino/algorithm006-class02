#include<iostream>
using namespace std;
int main()
{
	int year,month,day;
	int count = 0;
	int months[12] = { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
	cout << "请输入年月日" << endl;
	cin >> year>>month>>day;
	if (year % 4 == 0 && year % 100 != 0 || year % 400 == 0)
	{
		months[1] = 29;
	}
	for (int i = 0; i < month;i++)
	{
		for (int j = 0; j < months[i];j++)
		{
			count++;
			if (i==month-1&&j==day-1)
			{
				break;
			}
		}
	}
	cout << count;
	cin.get(); 
	cin.get();
	return 0;
}
