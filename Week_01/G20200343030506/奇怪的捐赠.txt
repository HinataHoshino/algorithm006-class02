#include<iostream>
#include <cmath>
using namespace std;
int cnt = 0;
int visit[10] = { 0 };
int num;
int b = 0;
void dfs(int step, int cnt)
{
	if (visit[step]>5 || cnt>1000000)//数量大于5数值大于1000000结束
	{
		return;
	}
	else if (cnt==1000000)//当数值为1000000时输出份数最多
	{
		num = 0;
		for (int i = 0; i < 10; i++)
		{
			num += visit[i];
		}
		if (b==0)
		cout << num<<endl;
		b++;
	}
	cnt += pow(7, step);
	visit[step]++;
	dfs(step, cnt);
	dfs(step + 1, cnt);
	visit[step]--;
	cnt -= pow(7, step);
}
int main()
{
	dfs(0, 0);
	cin.get();
	cin.get();
	return 0;
}
